CLI Commands

Partnership Management

# Register partner chain
cargo run -- roundtable register-partner --chain-id partner_001 --rpc-endpoint https://rpc.partner.com

# Create partnership agreement
cargo run -- roundtable create-partnership --partner-id partner_001 --revenue-share 20

# List active partnerships
cargo run -- roundtable list-partnerships

# Check partnership status
cargo run -- roundtable partnership-status --partner-id partner_001

Revenue Operations

# Check revenue distribution status
cargo run -- roundtable revenue-status

# Manual revenue distribution
cargo run -- roundtable distribute-revenue

# Treasury management
cargo run -- roundtable treasury-balance
cargo run -- roundtable treasury-operations

Integration Examples

use bpci_enterprise::round_table_oracle::RoundTableOracle;

// Initialize Round Table Oracle
let oracle = RoundTableOracle::new().await?;

// Register partner chain
let partner_config = PartnerChainConfig {
    chain_id: "partner_chain_001".to_string(),
    rpc_endpoint: "https://rpc.partner.com:8545".to_string(),
    websocket_endpoint: "wss://ws.partner.com:8546".to_string(),
    representative_address: "0x1234...abcd".to_string(),
    revenue_share_percentage: 20.0,
};

oracle.register_partner_chain(partner_config).await?;

// Create partnership agreement
let partnership = oracle.create_partnership(
    "partner_chain_001",
    partnership_terms
).await?;
// Roundtable Partnership SDK
import { RoundTableOracle } from '@bpci/roundtable-sdk';

// Initialize oracle
const oracle = new RoundTableOracle({
    endpoint: 'http://localhost:8081'
});

// Register partner chain
const partnerConfig = {
    chain_id: 'partner_chain_001',
    rpc_endpoint: 'https://rpc.partner.com:8545',
    websocket_endpoint: 'wss://ws.partner.com:8546',
    revenue_share_percentage: 20.0
};

await oracle.registerPartnerChain(partnerConfig);

// Check revenue distribution status
const revenueStatus = await oracle.getRevenueStatus();
console.log('Total distributed:', revenueStatus.total_revenue_distributed);
# Register partner chain
curl -X POST http://localhost:8081/api/roundtable/partners/register \
  -H "Content-Type: application/json" \
  -d '{
    "chain_id": "partner_chain_001",
    "rpc_endpoint": "https://rpc.partner.com:8545",
    "revenue_share_percentage": 20.0
  }'

# Get revenue distribution status
curl http://localhost:8081/api/roundtable/revenue/status

# Create partnership agreement
curl -X POST http://localhost:8081/api/roundtable/partnerships/create \
  -H "Content-Type: application/json" \
  -d '{
    "partner_chain_id": "partner_chain_001",
    "partnership_terms": {"revenue_share": 20.0}
  }'
