API Reference

Create Database Agreement

POST /api/cuedb/agreements
{
    "agreement_type": "enterprise",
    "wallet_id": "enterprise_wallet_123",
    "configuration": {
        "regions": ["us-east-1", "eu-west-1"],
        "replication_factor": 3,
        "encryption": "aes-256-gcm",
        "compliance": ["gdpr", "sox"]
    }
}

Execute Database Operation

POST /api/cuedb/operations
{
    "agreement_id": "agreement_123",
    "operation": "create_table",
    "parameters": {
        "table_name": "users",
        "schema": {
            "id": "uuid",
            "name": "string",
            "email": "string"
        }
    }
}

CLI Commands

Database Agreement Management

# Create database agreement
cargo run -- cue-db create-agreement --type enterprise

# List agreements
cargo run -- cue-db list-agreements

# Parse DBYML configuration
cargo run -- cue-db parse-dbyml --file schema.dbyml

# Execute database operation
cargo run -- cue-db execute-operation --operation create_table

Pipeline Operations

# Create ETL pipeline
cargo run -- cue-db create-pipeline --type etl

# Monitor pipeline status
cargo run -- cue-db pipeline-status --id pipeline_123

# Test system components
cargo run -- cue-db test-system

Integration Examples

use bpci_enterprise::cuedb::{CueDB, AgreementType};

// Create CueDB instance
let cuedb = CueDB::new().await?;

// Create enterprise agreement
let agreement = cuedb.create_agreement(
    AgreementType::Enterprise,
    "enterprise_wallet_123"
).await?;

// Execute database operation
let result = cuedb.execute_operation(
    &agreement.id,
    "create_table",
    serde_json::json!({
        "table_name": "users",
        "schema": {"id": "uuid", "name": "string"}
    })
).await?;
# CueDB Python SDK
from bpci_cuedb import CueDB, AgreementType

# Initialize CueDB
cuedb = CueDB(endpoint='http://localhost:8081')

# Create agreement
agreement = await cuedb.create_agreement(
    agreement_type=AgreementType.ENTERPRISE,
    wallet_id='enterprise_wallet_123'
)

# Execute operation
result = await cuedb.execute_operation(
    agreement_id=agreement.id,
    operation='create_table',
    parameters={
        'table_name': 'users',
        'schema': {'id': 'uuid', 'name': 'string'}
    }
)
# Create database agreement
curl -X POST http://localhost:8081/api/cuedb/agreements \
  -H "Content-Type: application/json" \
  -d '{
    "agreement_type": "enterprise",
    "wallet_id": "enterprise_wallet_123"
  }'

# Execute database operation
curl -X POST http://localhost:8081/api/cuedb/operations \
  -H "Content-Type: application/json" \
  -d '{
    "agreement_id": "agreement_123",
    "operation": "create_table",
    "parameters": {
      "table_name": "users",
      "schema": {"id": "uuid", "name": "string"}
    }
  }'
