CLI Commands

Node Management

# Register node
cargo run -- registry register-node --type community --endpoint https://node.example.com

# List nodes
cargo run -- registry list-nodes --registry bpi_community

# Get node info
cargo run -- registry node-info --node-id node_12345

# Update node status
cargo run -- registry update-node --node-id node_12345 --status active

Wallet Operations

# Stamp wallet
cargo run -- registry stamp-wallet --wallet-id wallet_123 --type bank

# List wallets
cargo run -- registry list-wallets --type government

# Verify wallet stamp
cargo run -- registry verify-stamp --wallet-id wallet_123

Integration Examples

use bpci_enterprise::registry::{BpciRegistry, NodeType};

// Initialize registry
let registry = BpciRegistry::new().await?;

// Register node
let node_config = NodeConfig {
    node_type: NodeType::Community,
    endpoint: "https://node.example.com:8080".to_string(),
    public_key: node_keypair.public_key(),
};

let registration = registry.register_node(node_config).await?;

// Get registry info
let info = registry.get_info().await?;
println!("Total nodes: {}", info.total_nodes);
// Registry SDK
import { BpciRegistry } from '@bpci/registry-sdk';

// Initialize registry
const registry = new BpciRegistry({
    endpoint: 'http://localhost:8081'
});

// Register node
const nodeConfig = {
    node_type: 'community',
    endpoint: 'https://node.example.com:8080',
    public_key: 'ed25519_public_key_hex'
};

const registration = await registry.registerNode(nodeConfig);

// Get registry information
const info = await registry.getInfo();
console.log('Active validators:', info.active_validators);
# Register node
curl -X POST http://localhost:8081/api/registry/nodes \
  -H "Content-Type: application/json" \
  -d '{
    "node_type": "community",
    "endpoint": "https://node.example.com:8080",
    "public_key": "ed25519_public_key_hex"
  }'

# Get registry info
curl http://localhost:8081/api/registry/info

# Stamp wallet
curl -X POST http://localhost:8081/api/registry/wallets/stamp \
  -H "Content-Type: application/json" \
  -d '{
    "wallet_id": "wallet_123",
    "stamp_type": "bank"
  }'
