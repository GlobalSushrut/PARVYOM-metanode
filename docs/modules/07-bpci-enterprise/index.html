CLI Commands

Platform Management

# Start BPCI Enterprise server
cargo run -- web start --port 8081

# Check economy status
cargo run -- economy status

# Registry operations
cargo run -- registry list-nodes
cargo run -- registry register-node --type community

# Wallet operations
cargo run -- wallet create --type normal
cargo run -- wallet stamp --type bank --wallet-id {wallet_id}

CueDB Operations

# Create database agreement
cargo run -- cue-db create-agreement --type enterprise

# Execute database operations
cargo run -- cue-db execute-operation --operation create_table

# Test CueDB system
cargo run -- cue-db test-system

Integration Examples

use bpci_enterprise::{BpciEnterprise, EconomyManager};

// Initialize BPCI Enterprise platform
let platform = BpciEnterprise::new().await?;

// Get economy status
let economy = platform.economy_manager();
let status = economy.get_status().await?;
println!("Total GEN supply: {}", status.total_supply.gen);

// Register new node
let registry = platform.registry();
registry.register_node("community", node_config).await?;
// BPCI Enterprise SDK
import { BpciEnterprise } from '@bpci/enterprise-sdk';

// Initialize platform
const platform = new BpciEnterprise({
    endpoint: 'http://localhost:8081'
});

// Get economy status
const economyStatus = await platform.economy.getStatus();
console.log('Active validators:', economyStatus.active_validators);

// Access stamped wallet APIs
const bankApi = platform.stamped.bank;
const settlement = await bankApi.settlement({
    wallet_id: 'bank_wallet_123',
    amount: '1000000'
});
# Get platform status
curl http://localhost:8081/api/status

# Get economy metrics
curl http://localhost:8081/api/economy/status

# Access bank stamped wallet API
curl -X POST http://localhost:8081/api/stamped/bank/settlement \
  -H "Authorization: Bearer {bank_token}" \
  -H "Content-Type: application/json" \
  -d '{
    "wallet_id": "bank_wallet_123",
    "operation": "settlement",
    "amount": "1000000"
  }'
